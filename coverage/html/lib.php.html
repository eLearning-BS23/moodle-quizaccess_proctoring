166function getMatchResult($refimageurl, $targetimageurl, $reportid): array

172    log_aws_api_call($reportid, $apiresponse);
173
174    // Update Match result.
175    if (!empty($similarityresult)) {
176        if (isset($similarityresult[0]['Similarity'])) {
177            $similarity = $similarityresult[0]['Similarity'];
178        } else {
179            $similarity = 0;
180            log_fm_warning($reportid);

183        $similarity = 0;
184        log_fm_warning($reportid);

187
188    return [$similarityresult, $similarity];
